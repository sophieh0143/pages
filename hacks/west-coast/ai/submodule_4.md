      const shuffled = [...arr].sort(() => Math.random() - 0.5);

          destinations: getRandomItems(data.destinations, 2),
          foods: getRandomItems(data.foods, 2),
          sports: getRandomItems(data.sports, 2)
        };
      });

      generateQuiz();
